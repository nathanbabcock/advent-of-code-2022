## Functional solution

```ts
function bufStart(buf: string, start = 0) {
  const maybeMarker = buf.slice(start, start + 4)
  const chars = new Set(...buf.split(''))
  if (chars.length === 4) return start + 4
  else return bufStart(buf.slice(1), start + 1)
}
```
